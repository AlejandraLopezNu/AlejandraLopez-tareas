Exercice 1
SELECT MAX(years_employed) as Max_years_employed
FROM employees;

Exercice 2
SELECT role,
 AVG(years_employed) as Average_years_employed
FROM employees
GROUP BY role;

Exercice 3
SELECT building, 
SUM(years_employed) as Total_years_employed
FROM employees
GROUP BY building;
